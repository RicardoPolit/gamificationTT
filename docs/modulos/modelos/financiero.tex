\subsection{Relaciones del módulo financiero}

    Esta sección no cuenta propiamente con entidades, sin embargo,
    se utilizan configuraciones que se guardan en la tabla de moodle 
    y dichas configuraciones se describen en esta sección.


    \begin{cdtEntidad}{tienda-configuraciones}{Configuraciones módulo financiero}{%
        Estas configuraciones llevan un control de cuántas monedas se reparten por
        cada tipo de actividad realizada.
        }

        \brAttr{victory-vs-system}{Victoria contra el sistema}{tInt}{%
            Cuántas monedas de plata de le darán a un usuario por vencer a
            el sistema de cada dificultad por primera vez.\par

            \it Restricciones:
            \refElem{tDefault} $: 50$, 
            \refElem{tRequired}.
        }

        \brAttr{victory-vs-user}{Victoria contra compañero}{tInt}{%
            Cuántas monedas de plata se le darán a un usuario cuando venza a otro usuario.\par

            \it Restricciones:
            \refElem{tDefault} $: 50$, 
            \refElem{tRequired}.
        }

        \brAttr{daily-question}{Pregunta diaria respondida}{tInt}{%
            Cuántas monedas de plata se le darán a un usurio cuando responda una pregunta diaria.\par

            \it Restricciones:
            \refElem{tDefault} $: 50$, 
            \refElem{tRequired}.
        }
        \brAttr{victory-vs-system-enabled}{Habilitar victoria contra el sistema}{tBoolean}{%
            Bandera que indica si está habilitado el dar monedas cuando se gane al sistema.\par

            \it Restricciones:
            \refElem{tDefault} $: true$, 
            \refElem{tRequired}.
        }

        \brAttr{victory-vs-user-enabled}{Habilitar victoria contra compañero}{tBoolean}{%
            Bandera que indica si está habilitado el dar monedas cuando se gane a otro usuario.\par

            \it Restricciones:
            \refElem{tDefault} $: true$, 
            \refElem{tRequired}.
        }

        \brAttr{daily-question-enabled}{Habilitar pregunta diaria respondida}{tBoolean}{%
            Bandera que indica si está habilitado el dar monedas cuando responda una pregunta diaria.\par

            \it Restricciones:
            \refElem{tDefault} $: true$, 
            \refElem{tRequired}.
        }
    \end{cdtEntidad}
