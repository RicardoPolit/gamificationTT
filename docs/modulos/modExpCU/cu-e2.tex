\begin{UCtrayectoria}

    \Actor Presiona el botón \IUMenu en la esquina superior izquierda de la pantalla para abrir el
           menu desplegable

    \Actor Selecciona la opción: \IUAdminSitio Administración del sitio

    \Sistema Carga la pantalla \refElem{IU-MAdminSitio}

    \Actor Selecciona ver las opciones para administrar plugins presionando la pestaña
           {\it Plugins}

    \Sistema Obtiene las categorias de las opciones que se muestran en la pantalla.

    \Sistema En la categoría {\it Bloques} agrega la subcategoría \salida{GamedleLevel} con los
             enlaces a las configuraciones \salida{ConfExpGral}, \salida{ConfExpComp} y
             \salida{ConfExpVisual}. \refErr{Uno}

    \Sistema Carga la pantalla \refElem{IU-MAdminSitio-Plugins}
